class.EventObserver

Since

1.62.0

Example

ts
import { EventObserver } from 'super-tools-lib'

const _eventPool = new EventObserver()

const click = () => {}
_eventPool.on('myclick', click)

// 触发已经注册好的myclick事件
_eventPool.spread('myclick', {name:1})
